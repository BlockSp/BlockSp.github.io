390 ///Calculate the determinant of a TinyMatrix.
391 ///Warning! Explicit determinant calculations are an O(N!), where ! is factorial.
392 ///Use for small N only.
393 template<typename T, int N>
394 T determinant(const blitz::TinyMatrix<T, N, N>& A)
395 {
396 if constexpr (N == 1) return A(0, 0);
397 if constexpr (N == 2) return A(0, 0) * A(1, 1) - A(1, 0) * A(0, 1);
398 else
399 {
400 T det = 0.0;
401 for (int j = 0; j < N; ++j) //scan across column of the first row
402 {
403 blitz::TinyMatrix<T, N - 1, N - 1> ASub; //make submatrix
404 for (int is = 0; is < N - 1; ++is)
405 for (int js = 0; js < N; ++js) if (js != j)
406 {
407 int col_s = js > j ? js - 1 : js; //find column for ASub
408 ASub(is, col_s) = A(is + 1, js);
409 }
410 T val = A(0, j) * determinant<T, N - 1>(ASub); //recursive call to submatrix
411 if (j % 2 == 1) val *= -1; //include the -1 every other column
412 det += val;
413 }
414 return det;
415 }

838 ///Compute explicit inverse of a small 1x1, 2x2, and 3x3 matrix
839 ///A^-1 = 1/det * adj(A)
840 ///det = deteminant of A, adj(A) = Adjoint matrix of A.
841 template<typename T, int N, std::enable_if_t<N <= 3, bool> = true>
842 blitz::TinyMatrix<T, N, N> explicit_small_Inv(const blitz::TinyMatrix<T, N, N>& A)
843 {
844 if constexpr (N == 1) return blitz::TinyMatrix<T, N, N>(1.0 / A(0, 0));
845 else if constexpr (N == 2)
846 {
847 //A = [a b] , A^-1 = 1/(ad - bc)[d -b]
848 // [c d] [-c a]
849 T det = A(0, 0) * A(1, 1) - A(0, 1) * A(1, 0); //Determinant of A
850 blitz::TinyMatrix<T, N, N> A_Inv;
851 A_Inv(0, 0) = A(1, 1);
852 A_Inv(0, 1) = -A(0, 1);
853 A_Inv(1, 0) = -A(1, 0);
854 A_Inv(1, 1) = A(0, 0);
855 A_Inv /= det;
856 return A_Inv;
857 }
858 else if constexpr (N == 3)
859 {
860 blitz::TinyMatrix<T, N, N> A_Inv;
861
862 //1) Explicitly compute Adjoint matrix of A. i.e. adj(A)
863 //This is equal to the transpose of the Cofactor matrix.
864 A_Inv(0, 0) = A(1, 1) * A(2, 2) - A(2, 1) * A(1, 2);
865 A_Inv(1, 0) = -(A(1, 0) * A(2, 2) - A(2, 0) * A(1, 2));
866 A_Inv(2, 0) = A(1, 0) * A(2, 1) - A(2, 0) * A(1, 1);
867
868 A_Inv(0, 1) = -(A(0, 1) * A(2, 2) - A(2, 1) * A(0, 2));
869 A_Inv(1, 1) = A(0, 0) * A(2, 2) - A(2, 0) * A(0, 2);
870 A_Inv(2, 1) = -(A(0, 0) * A(2, 1) - A(2, 0) * A(0, 1));
871
872 A_Inv(0, 2) = A(0, 1) * A(1, 2) - A(1, 1) * A(0, 2);
873 A_Inv(1, 2) = -(A(0, 0) * A(1, 2) - A(1, 0) * A(0, 2));
874 A_Inv(2, 2) = A(0, 0) * A(1, 1) - A(1, 0) * A(0, 1);
875
876 //2) Calculate determinant of matrix A
877 T det = A(0, 0) * (A(1, 1) * A(2, 2) - A(1, 2) * A(2, 1))
878 - A(0, 1) * (A(1, 0) * A(2, 2) - A(1, 2) * A(2, 0))
879 + A(0, 2) * (A(1, 0) * A(2, 1) - A(1, 1) * A(2, 0));
880
881 //3) Return A^-1 = 1/det * adj(A)
882 A_Inv /= det;
883 return A_Inv;
884 }
885 else
886 {
887 std::cout << "N = " << N << " is too large for explicit_small_Inv. A returned" << std::endl;
888 return A;
889 }
890 }
891
892 ///Compute explicit inverse of a small symmetic matrix A stored in compact vector form.
893 ///A length N*(N+1)/2, stored as if spanning the LOWER triangular portion of a matrix.
894 ///A^-1 = 1/det * adj(A)
895 ///det = deteminant of A, adj(A) = Adjoint matrix of A.
896 ///For N = 1, 2, and 3 only.
897 template<typename T, int N, std::enable_if_t<N <= 3, bool> = true>
898 blitz::TinyMatrix<T, N, N> explicit_small_Inv(const blitz::TinyVector<T, (N* (N + 1)) / 2>& A)
899 {
900 if constexpr (N == 1) return blitz::TinyMatrix<T, 1, 1>(1.0 / A(0));
901 else if constexpr (N == 2)
902 {
903 //A = [a b] , A^-1 = 1/(ad - bc)[d -b]
904 // [c d] [-c a]
905 T det = A(0) * A(2) - A(1) * A(1); //Determinant of A
906 blitz::TinyMatrix<T, N, N> A_Inv;
907 A_Inv(0, 0) = A(2);
908 A_Inv(0, 1) = -A(1);
909 A_Inv(1, 0) = -A(1);
910 A_Inv(1, 1) = A(0);
911 A_Inv /= det;
912 return A_Inv;
913 }
914 else if constexpr (N == 3)
915 {
916 blitz::TinyMatrix<T, 3, 3> A_Inv;
917
918 //1) Explicitly compute Adjoint matrix of A. i.e. adj(A)
919 //This is equal to the transpose of the Cofactor matrix.
920 A_Inv(0, 0) = A(2) * A(5) - A(4) * A(4);
921 A_Inv(1, 0) = -(A(1) * A(5) - A(3) * A(4));
922 A_Inv(2, 0) = A(1) * A(4) - A(3) * A(2);
923 A_Inv(1, 1) = A(0) * A(5) - A(3) * A(3);
924 A_Inv(2, 1) = -(A(0) * A(4) - A(3) * A(1));
925 A_Inv(2, 2) = A(0) * A(2) - A(1) * A(1);
926 A_Inv(0, 1) = A_Inv(1, 0);
927 A_Inv(0, 2) = A_Inv(2, 0);
928 A_Inv(1, 2) = A_Inv(2, 1);
929
930 //2) Calculate determinant of matrix A
931 T det = A(0) * (A(2) * A(5) - A(4) * A(4))
932 - A(1) * (A(1) * A(5) - A(4) * A(3))
933 + A(3) * (A(1) * A(4) - A(2) * A(3));
934
935 //3) Return A^-1 = 1/det * adj(A)
936 A_Inv /= det;
937 return A_Inv;
938 }
939 else
940 {
941 std::cout << "N = " << N << " is too large for explicit_small_Inv. A returned" << std::endl;
942 return A;
943 }
944 }
