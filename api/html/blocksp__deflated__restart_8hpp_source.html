13 ///Restart pGMRES with a size k+1 Krylov subspace that consists of k harmonic Ritz vectors.
14 ///Harmonic Ritz vectors are picked to approximate the smallest eigenvalues/vectors.
15 ///These are the eigenvalues that give pGMRES the most difficulty.
16 ///Having these vectors in the restart space "deflates" the space pGMRES must search,
17 ///leading to better performance.
18 ///Non-restarted pGMRES may become intractably large, pure restarts may stall.
19 ///See "GMRES with Deflated Restarting" by Ronald B. Morgan for more details.
20 ///
21 ///Note: this implementation of deflated restart pGMRES may still have issues.
22 ///For some problems, the pGMRES iteration count greatly increases and is slower than non-restarted.
23 ///For others, the increase is minimal and the deflated restart version is faster than non-restarted.
24 ///There are likely some issues with numerical stability.
25 ///Try pGMRES with and without restarts, see which works best for your problem.
26 ///BlockSp offers SOR and ILU0 pGMRES with and without restarting.
27 ///
28 ///There are a few commented out debugging protocols at the bottom of this file
29 ///If you find any bugs, please let me know :)
30 ///
31 ///This program inputs the (n+1) by n upper Hessenberg Arnoldi matrix H,
32 ///the orthonormal Krylov basis Qn, the residual vector res_vec,
33 ///the residual coefficient vector c_r, the residual magnitudes res_mag,
34 ///and the restart space size k. Each is modified during the restart.
35 ///Works for both reals and complex.
36 ///
37 ///AI disclaimer: deflated_restart was written with assistance from Anthropic's Claude Opus 4.5.
38 template<typename T, int B, int D>
39 void deflated_restart(blitz::Array<T, 2>& H, std::vector<solArray<T, B, D>>& Qn,
40 blitz::Array<T, 1>& res_vec, blitz::Array<T, 1>& c_r,
41 std::vector<ScalarType_t<T>>& res_mag, int k)
42 {
43 using S = ScalarType_t<T>;
44 int m = H.extent(0); //number of rows of H, m = n + 1,
45 int n = H.extent(1); //number of columns of H
46 if (k > n) k = n; //limit k to available eigenvalues
47 S zero_thresh = 10 * n * n * std::numeric_limits<S>::epsilon();
48
49 if (k == 0) //pure restart
50 {
51 H.resize(0);
52 Qn.clear();
53 res_mag.clear();
54 c_r.resize(1);
55 return;
56 }
57
58 //////////////////////////////////////////////////////
59 ///(1) Compute H_tilde = H_n + |h_mn|^2 * H_n^{-T} * e_n * e_n^T
60 ///The eigenvalues of H_tilde are the harmonic Ritz values.
61 ///H_n is the nxn square portion of H.
62 ///h_mn the bottom right sub-diagonal element of H.
63 ///H_n^{-T} is the inverse of the transpose of H_n.
64 ///e_n is the n-th standard basis vector.
65
66 //Extract square portion H_n (n x n)
67 blitz::Array<T, 2> H_n(n);
68 for (int i = 0; i < n; ++i)
69 for (int j = 0; j < n; ++j)
70 H_n(i, j) = H(i, j);
71
72 //Get h_mn, the bottom right sub-diagonal element of H
73 T h_mn = H(n, n - 1);
74
75 //Solve H_n^T * w = e_n for w (compute H_n^{-T} * e_n), via LU factorization.
76 blitz::Array<T, 1> w(n); w = T(0);
77 w(n - 1) = T(1); // e_n
78 {
79 dense::LUFactor_Arr<T> H_nT_LU(dense::transpose(H_n));
80 H_nT_LU.invmul(w);
81 }
82
83 //Form H_tilde (stored in H_n) = H_n + |h_mn|^2 * w * e_n^T
84 //The second rhs term only modifies the last column of H_n
85 T scale;
86 if constexpr (is_complex_v<T>) scale = std::conj(h_mn) * h_mn;
87 else scale = h_mn * h_mn;
88 for (int i = 0; i < n; ++i)
89 H_n(i, n - 1) += scale * w(i);
90
91 //////////////////////////////////////////////////////
92 ///(2) Compute the Schur decomposition of H_tilde
93 ///Then reorder to put the k smallest eigenvalues first
94
95 blitz::Array<T, 2> Z(n, n); //Schur vectors
96 lapack_int sdim = 0;
97 if constexpr (is_complex_v<T>) //Complex case
98 {
99 //Compute for eigenvalues and Schur factorization
100 blitz::Array<T, 1> eig_vals(n);
101 lapack::gees<T>(LAPACK_ROW_MAJOR, 'V', 'N', nullptr, n,
102 H_n.data(), n, &sdim, eig_vals.data(), Z.data(), n);
103
104 //Create a permutation function to find smallest Ritz values
105 auto compare = [&](const auto& a, const auto& b) { return std::abs(a) < std::abs(b); };
106 auto perm{ util::sort_permutation(eig_vals, compare) };
107
108 //Build select array for k smallest
109 std::vector<lapack_logical> select(n, 0);
110 for (int i = 0; i < k; ++i)
111 select[perm[i]] = 1;
112
113 //Reorder Schur form, sort eigenvalues by magnitude to find k smallest
114 S s_out, sep_out;
115 lapack::trsen<T>(LAPACK_ROW_MAJOR, 'V', 'V', select.data(), n,
116 H_n.data(), n, Z.data(), n, eig_vals.data(), &sdim, &s_out, &sep_out);
117 }
118 else //Real case
119 {
120 blitz::Array<T, 1> wr(n), wi(n); //real and imaginary parts of eigenvalues
121 lapack::gees<T>(LAPACK_ROW_MAJOR, 'V', 'N', nullptr, n,
122 H_n.data(), n, &sdim, wr.data(), wi.data(), Z.data(), n);
123
124 //Sort eigenvalues by magnitude to find k smallest
125 std::vector<int> perm(n);
126 std::iota(perm.begin(), perm.end(), 0);
127 std::sort(perm.begin(), perm.end(), [&](int i, int j) {
128 S mag_1 = std::sqrt(wr(i) * wr(i) + wi(i) * wi(i));
129 S mag_2 = std::sqrt(wr(j) * wr(j) + wi(j) * wi(j));
130 return mag_1 < mag_2;
131 });
132
133 //Build the select array for k smallest eigenvalues.
134 //If selecting a complex eigenvalue that's part of a 2x2 block,
135 //both real and imaginary parts of the eigenvalue are selected.
136 std::vector<lapack_logical> select(n, 0);
137 int cnt = 0;
138 for (int i = 0; i < n && cnt < k; ++i)
139 {
140 int ip = perm[i];
141 if (select[ip]) continue; //already selected as part of a pair
142 select[ip] = 1;
143 cnt++;
144
145 //If this is a complex eigenvalue (wi != 0),
146 //find and select its conjugate pair, at ip+1 or ip-1.
147 if (std::abs(wi(ip)) > zero_thresh)
148 {
149 //Find the conjugate
150 for (int j = 0; j < n; ++j)
151 {
152 if (j != ip &&
153 std::abs(wr(j) - wr(ip)) < zero_thresh &&
154 std::abs(wi(j) + wi(ip)) < zero_thresh)
155 {
156 if (!select[j])
157 {
158 select[j] = 1;
159 cnt++;
160 }
161 break;
162 }
163 }
164 }
165 }
166
167 //Reorder Schur form to put selected eigenvalues first
168 S s_out = 0; S sep_out = 0;
169 lapack::trsen<T>(LAPACK_ROW_MAJOR, 'V', 'V', select.data(), n,
170 H_n.data(), n, Z.data(), n, wr.data(), wi.data(), &sdim, &s_out, &sep_out);
171 }
172
173 //Update k, which may have changed in real systems due to complex eigenvalues
174 k = sdim;
175
176 //////////////////////////////////////////////////////
177 ///(3) Extract P_k (first k Schur vectors) and orthonormalize
178
179 //P_k is the first k columns of Z
180 blitz::Array<T, 2> P_k(n, k);
181 for (int i = 0; i < n; ++i)
182 for (int j = 0; j < k; ++j)
183 P_k(i, j) = Z(i, j);
184
185 //Orthonormalize P_k via QR factorization
186 //The Schur vectors are orthonormal, but this aids numerical stability
187 {
188 blitz::Array<T, 1> tau(k);
189 lapack::geqrf<T>(LAPACK_ROW_MAJOR, n, k, P_k.data(), k, tau.data());
190 if constexpr (is_complex_v<T>)
191 lapack::ungqr<T>(LAPACK_ROW_MAJOR, n, k, k, P_k.data(), k, tau.data());
192 else lapack::orgqr<T>(LAPACK_ROW_MAJOR, n, k, k, P_k.data(), k, tau.data());
193 }
194
195 //////////////////////////////////////////////////////
196 ///(4) Form P_{k+1} by extending P_k and orthonormalizing res_vec
197
198 //P_{k+1} is (n+1)x(k+1)
199 blitz::Array<T, 2> P_kp1(n + 1, k + 1);
200 P_kp1 = T(0);
201
202 //Copy P_k into first k columns (with zero in last row)
203 for (int i = 0; i < n; ++i)
204 for (int j = 0; j < k; ++j)
205 P_kp1(i, j) = P_k(i, j);
206
207 //Put res_vec in last column
208 for (int i = 0; i < n + 1; ++i)
209 P_kp1(i, k) = res_vec(i);
210
211 //Gram-Schmidt: orthogonalize last column against first k
212 {
213 for (int i = 0; i < k; ++i)
214 {
215 T d = T(0);
216 for (int j = 0; j < n + 1; ++j)
217 if constexpr (is_complex_v<T>) d += std::conj(P_kp1(j, i)) * P_kp1(j, k);
218 else d += P_kp1(j, i) * P_kp1(j, k);
219 for (int j = 0; j < n + 1; ++j)
220 P_kp1(j, k) -= d * P_kp1(j, i);
221 }
222 //Normalize last column
223 T norm = T(0);
224 for (int j = 0; j < n + 1; ++j)
225 if constexpr (is_complex_v<T>) norm += std::conj(P_kp1(j, k)) * P_kp1(j, k);
226 else norm += P_kp1(j, k) * P_kp1(j, k);
227 norm = std::sqrt(norm);
228 for (int j = 0; j < n + 1; ++j)
229 P_kp1(j, k) /= norm;
230 }
231
232 //////////////////////////////////////////////////////
233 ///(5) Compute new residual coefficient vector c_r = P_{k+1}^T * res_vec
234
235 c_r.resize(k + 1);
236 c_r = T(0);
237 for (int i = 0; i < k + 1; ++i)
238 for (int j = 0; j < n + 1; ++j)
239 if constexpr(is_complex_v<T>) c_r(i) += std::conj(P_kp1(j, i)) * res_vec(j);
240 else c_r(i) += P_kp1(j, i) * res_vec(j);
241
242 //////////////////////////////////////////////////////
243 ///(6) Form the new H: H_new = P_{k+1}^T * H * P_k
244
245 auto H_new{ dense::matMat_transpose(P_kp1, dense::matMat(H, P_k)) };
246
247 //////////////////////////////////////////////////////
248 ///(7) Form the new basis: Qn_new = Qn * P_{k+1}
249
250 int len = Qn[0].len();
251 std::vector<solArray<T, B, D>> Qn_new(k + 1);
252 for (int i = 0; i < k + 1; ++i)
253 Qn_new[i].resize(len);
254
255 for (int im = 0; im < m; ++im)
256 for (int ik = 0; ik < k + 1; ++ik)
257 for (int il = 0; il < len; ++il)
258 Qn_new[ik](il) += Qn[im](il) * P_kp1(im, ik);
259
260 //////////////////////////////////////////////////////
261 ///(8) Set new values
262
263 H.resize(H_new.extent());
264 H = H_new;
265 Qn = Qn_new;
266 std::vector<S> res_mag_new;
267 res_mag_new.push_back(res_mag.back());
268 res_mag = res_mag_new;
269 }
