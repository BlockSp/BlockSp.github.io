13///blocksp_containers.hpp defines the fundamental containers of BlockSp,
14///a templated C++ library for block sparse linear algebra.
15///These include:
16///
17///Sp::CSR - General/templated Condensed Row Storage (CSR) sparse matrix.\n
18///BlockSp - Block CSR Matrix.\n
19///solArray - Multi-dimension block Solution Array/Vector.\n
20///
21///Notation: B = block size of solArray, D = dimension of solArray. \n
22///BlockSp matrices typically have block size of B or B*D.
23///B = D = 1 reduces to standard sparse linear algebra. \n
24///Data type T, block size B, and dimension D are compile-time template parameters,
25///resulting in fast, optimized operations.
26///
27///This file also includes dot product, infinity norm, magnitude, and two norm functions for solArrays. \n
28///And also includes fundamental types ScalarType, ComplexType, and is_complex. \n
29///BlockSp matrix-matrix (bsmm) and matrix-vector (bsmv) multiplication are in blocksp_multiply.hpp. \n
30///Preconditioned conjugate gradient, GMRES, and Arnoldi eigenvalue solvers are in blocksp_solvers.hpp. \n
31///The dense linear algebra routines, e.g. multiplication and solvers, are in blocksp_dense_multiply/solvers.hpp. \n
32///Preconditioners are in blocksp_preconditioners.hpp.

45 ///@typedef BlockSp - a Sp::CSR matrix with dense square BxB
46 /// sized matrices/blocks as elements, i.e., the A in Ax = b.
47 ///Dense matrix data type templated on T.
48 ///The block size B is specified at compile time via template
49 /// for fast optimized stack-based operations.
50 ///Note: for problems with a D-dimension solArray<T,B,D>,
51 /// matrix block sizes are often B*D.
52 template<typename T, int B>
53 using BlockSp = Sp::CSR<blitz::TinyMatrix<T, B, B>>;

317 CSR<T> transpose() const

320 for (int im = 0; im < m_m; im++)

322 if constexpr (is_complex_v<T>) trans(m_entries[im][in].colInd(), im) = std::conj(m_entries[im][in].data());
323 else trans(m_entries[im][in].colInd(), im) = m_entries[im][in].data();

585 ///Solution Arrays (solArray) are the block vectors used by BlockSp.
586 ///i.e. the x and b in Ax = b.
587 ///solArrays have arbitrary dimension and block size.
588 ///They may be used to represent multi-dimensional solutions, e.g. for high-order finite element problems.
589 ///For example, the velocity and pressure variables within a 3D Navier-Stokes system (u,v,w,p)
590 ///can be stored in a single solArray.
591 ///
592 ///Structured as blitz::Array<TinyVector<TinyVector<T, Blocksize>, Dimension>, 1>,
593 ///the length of the solArray is equal to m_m or m_n of the BlockSp matrix, i.e., number of rows or columns.
594 ///Within each "element" of a solArray is a Dimension TinyVector that contains the Block TinyVectors.
595 ///solArray u is indexed as u(ie, dim)(ib) when D > 1 and u(ie)(ib) when D=1 (default).
596 ///ie = element index, dim = dimension index, ib = block index.
597 ///
598 ///For example, a solArray u of length 4, with B=3, D=2 will look like: u = \n
599 /// [ ({x1,x2,x3},{y1,y2,y3}) ] \n
600 /// [ ({x1,x2,x3},{y1,y2,y3}) ] \n

950 ///Take the infinity norm of a solArray.
951 ///Along all dimensions if dim < 0 (default), specific dimension if dim >= 0
952 template<typename T, int B, int D = 1>
953 inline ScalarType_t<T> infNorm(const solArray<T, B, D>& sa, int dim = -1)
954 {
955 assert(dim < D);
956 ScalarType_t<T> norm = 0;
957 for (const auto& el : sa)
958 {
959 if (dim >= 0) norm = std::max(norm, blitz::max(blitz::abs(el(dim))));
960 else for (int dim2 = 0; dim2 < D; ++dim2)
961 norm = std::max(norm, blitz::max(blitz::abs(el(dim2))));
962 }
963 return norm;
964 }
965
966 ///Take the infinity norm of a solArray along the specified dimensions.
967 template<typename T, int B, int D = 1>
968 inline ScalarType_t<T> infNorm(const solArray<T, B, D>& sa, const std::vector<int>& dims)
969 {
970 ScalarType_t<T> norm = 0;
971 for (const auto& dim : dims)
972 {
973 assert(dim >= 0 && dim < D);
974 norm = std::max(norm, infNorm<T, B, D>(sa, dim));
975 }
976 return norm;
977 }

979 ///Magnitude of solArray at each block point
980 ///i.e. - mag(x(im)) = (|x(im, dim1)|^2 + ... + |x(im, dimD)|)^2)^0.5
981 template<typename T, int B, int D = 1>
982 solArray<ScalarType_t<T>, B> mag(const solArray<T, B, D>& sa)
983 {
984 solArray<ScalarType_t<T>, B> magn(sa.len());
985 for (int ie = 0; const auto& el : sa)
986 {
987 blitz::TinyVector<ScalarType_t<T>, B> magB(0.0);
988 for (int dim = 0; dim < D; ++dim)
989 for (int ib = 0; ib < B; ++ib)
990 {
991 T val = el(dim)(ib);
992 if constexpr (is_complex_v<T>)
993 {
994 T val_sqr = std::conj(val) * val;
995 magB(ib) += val_sqr.real();
996 }
997 else magB(ib) += val * val;
998 }
999 for (int ib = 0; ib < B; ++ib)
1000 magB(ib) = std::sqrt(magB(ib));
1001 magn(ie) = magB;
1002 ie++;
1003 }
1004 return magn;
1005 }

1007 ///Dot product of two solArrays.
1008 ///Along all dimensions if dim < 0 (default), specific dimension if dim >= 0.
1009 template<typename T, int B, int D = 1>
1010 T dot(const solArray<T, B, D>& x1, const solArray<T, B, D>& x2, int dim = -1)
1011 {
1012 assert(dim < D);
1013 assert(x1.len() == x2.len());
1014 T dot = 0.0;
1015 for (int ie = 0; ie < x1.len(); ++ie)
1016 {
1017 if constexpr (is_complex_v<T>) //complex dot product requires complex conjugation
1018 {
1019 if (dim >= 0) dot += blitz::dot(blitz::conj(x1(ie, dim)), x2(ie, dim));
1020 else for (int dim2 = 0; dim2 < D; ++dim2)
1021 dot += blitz::dot(blitz::conj(x1(ie, dim2)), x2(ie, dim2));
1022 }
1023 else
1024 {
1025 if (dim >= 0) dot += blitz::dot(x1(ie, dim), x2(ie, dim));
1026 else for (int dim2 = 0; dim2 < D; ++dim2)
1027 dot += blitz::dot(x1(ie, dim2), x2(ie, dim2));
1028 }
1029 }
1030 return dot;
1031 }

1034 ///Along all dimensions if dim < 0, specific dimension if dim >= 0.

1036 inline ScalarType_t<T> twoNorm(const solArray<T, B, D>& x, const solArray<T, B, D>& y, int dim = -1)
