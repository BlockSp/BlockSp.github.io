13///blocksp_containers.hpp defines the fundamental containers of BlockSp,
14///a templated C++ library for block sparse linear algebra.
15///These include:
16///
17///Sp::CSR - General/templated Condensed Row Storage (CSR) sparse matrix.\n
18///BlockSp - Block CSR Matrix.\n
19///solArray - Multi-dimension block Solution Array/Vector.\n
20///
21///Notation: B = block size of solArray, D = dimension of solArray. \n
22///BlockSp matrices typically have block size of B or B*D.
23///B = D = 1 reduces to standard sparse linear algebra. \n
24///Data type T, block size B, and dimension D are compile-time template parameters,

45 ///@typedef BlockSp - a Sp::CSR matrix with dense square BxB
46 /// sized matrices/blocks as elements, i.e., the A in Ax = b.
47 ///Dense matrix data type templated on T.
48 ///The block size B is specified at compile time via template
49 /// for fast optimized stack-based operations.
50 ///Note: for problems with a D-dimension solArray<T,B,D>,
51 /// matrix block sizes are often B*D.
52 template<typename T, int B>

550 ///Solution Arrays (solArray) are the block vectors used by BlockSp.
551 ///i.e. the x and b in Ax = b.
552 ///solArrays have arbitrary dimension and block size.
553 ///They may be used to represent multi-dimensional solutions, e.g. for high-order finite element problems.
554 ///For example, the velocity and pressure variables within a 3D Navier-Stokes system (u,v,w,p)
555 ///can be stored in a single solArray.
556 ///
557 ///Structured as blitz::Array<TinyVector<TinyVector<T, Blocksize>, Dimension>, 1>,
558 ///the length of the solArray is equal to m_m or m_n of the BlockSp matrix, i.e., number of rows or columns.
559 ///Within each "element" of a solArray is a Dimension TinyVector that contains the Block TinyVectors.
560 ///solArray u is indexed as u(ie, dim)(ib) when D > 1 and u(ie)(ib) when D=1 (default).
561 ///ie = element index, dim = dimension index, ib = block index.
562 ///
563 ///For example, a solArray u of length 4, with B=3, D=2 will look like: u = \n
564 /// [ ({x1,x2,x3},{y1,y2,y3}) ] \n
565 /// [ ({x1,x2,x3},{y1,y2,y3}) ] \n
566 /// [ ({x1,x2,x3},{y1,y2,y3}) ] \n
567 /// [ ({x1,x2,x3},{y1,y2,y3}) ]
568 ///
569 ///Notes: setting B = D = 1, reduces the problem to regular sparse matrix-vector operations.
570 ///Block sparse matrix-vector multiplication can be performed along each dimension seperately or all together.
571 ///The use of TinyVectors as blocks/dimensions allows for fast stack-based compile-time optimized operations.
572 template<typename T, int B, int D = 1>
573 class solArray
574 {
575 private:
576
577 /////////////////////////

579 /////////////////////////

915 ///Take the infinity norm of a solArray.
916 ///Along all dimensions if dim < 0 (default), specific dimension if dim >= 0
917 template<typename T, int B, int D = 1>
918 inline ScalarType_t<T> infNorm(const solArray<T, B, D>& sa, int dim = -1)
919 {
920 assert(dim < D);
921 ScalarType_t<T> norm = 0;
922 for (const auto& el : sa)
923 {
924 if (dim >= 0) norm = std::max(norm, blitz::max(blitz::abs(el(dim))));
925 else for (int dim2 = 0; dim2 < D; ++dim2)
926 norm = std::max(norm, blitz::max(blitz::abs(el(dim2))));
927 }
928 return norm;
929 }
930
931 ///Take the infinity norm of a solArray along the specified dimensions.
932 template<typename T, int B, int D = 1>
933 inline ScalarType_t<T> infNorm(const solArray<T, B, D>& sa, const std::vector<int>& dims)
934 {
935 ScalarType_t<T> norm = 0;
936 for (const auto& dim : dims)
937 {
938 assert(dim >= 0 && dim < D);
939 norm = std::max(norm, infNorm<T, B, D>(sa, dim));
940 }
941 return norm;
942 }

944 ///Magnitude of solArray at each block point
945 ///i.e. - mag(x(im)) = (|x(im, dim1)|^2 + ... + |x(im, dimD)|)^2)^0.5
946 template<typename T, int B, int D = 1>
947 solArray<ScalarType_t<T>, B> mag(const solArray<T, B, D>& sa)
948 {
949 solArray<ScalarType_t<T>, B> magn(sa.len());
950 for (int ie = 0; const auto& el : sa)
951 {
952 blitz::TinyVector<ScalarType_t<T>, B> magB(0.0);
953 for (int dim = 0; dim < D; ++dim)
954 for (int ib = 0; ib < B; ++ib)
955 {
956 T val = el(dim)(ib);
957 if constexpr (is_complex_v<T>)
958 {
959 T val_sqr = std::conj(val) * val;
960 magB(ib) += val_sqr.real();
961 }
962 else magB(ib) += val * val;
963 }
964 for (int ib = 0; ib < B; ++ib)
965 magB(ib) = std::sqrt(magB(ib));
966 magn(ie) = magB;
967 ie++;
968 }
969 return magn;
970 }

972 ///Dot product of two solArrays.
973 ///Along all dimensions if dim < 0 (default), specific dimension if dim >= 0.
974 template<typename T, int B, int D = 1>
975 T dot(const solArray<T, B, D>& x1, const solArray<T, B, D>& x2, int dim = -1)
976 {
977 assert(dim < D);
978 assert(x1.len() == x2.len());
979 T dot = 0.0;
980 for (int ie = 0; ie < x1.len(); ++ie)
981 {
982 if constexpr (is_complex_v<T>) //complex dot product requires complex conjugation
983 {
984 if (dim >= 0) dot += blitz::dot(blitz::conj(x1(ie, dim)), x2(ie, dim));
985 else for (int dim2 = 0; dim2 < D; ++dim2)
986 dot += blitz::dot(blitz::conj(x1(ie, dim2)), x2(ie, dim2));
987 }
988 else
989 {
990 if (dim >= 0) dot += blitz::dot(x1(ie, dim), x2(ie, dim));
991 else for (int dim2 = 0; dim2 < D; ++dim2)
992 dot += blitz::dot(x1(ie, dim2), x2(ie, dim2));
993 }
994 }
995 return dot;
996 }

998 ///Two norm function for two solArrays, i.e. square root of dot product: sqrt(x dot y).
999 ///Along all dimensions if dim < 0, specific dimension if dim >= 0.
1000 template<typename T, int B, int D = 1>
1001 inline ScalarType_t<T> twoNorm(const solArray<T, B, D>& x, const solArray<T, B, D>& y, int dim = -1)
