192 ///Symmetric eigenvalue factorization - for real, symmetric NxN matrix: \n
193 ///A = QDQ^T, A^-1 = QD^-1Q^T. Stores A^-1 and D. \n
194 ///Regularizing: removes near zero eigen values.
195 ///May be used for inverting block diagonals in symmetric SOR.
196 template<typename T, int N> requires (!is_complex_v<T>)
197 struct SymEigenFactor
198 {
199 blitz::TinyVector<T, N> D;
200 blitz::TinyMatrix<T, N, N> AInv;
201 T tolFactor;
202
203 SymEigenFactor() {}
204
205 SymEigenFactor(blitz::TinyMatrix <T, N, N> A, T tolFactor = 1.0) :
206 tolFactor{ tolFactor }
207 {
208 //eigen decomp
209 lapack_int m;
210 blitz::TinyVector <lapack_int, N * 2> supp;
211 blitz::TinyMatrix<T, N, N> Q;
212 lapack::syevr<T>(LAPACK_ROW_MAJOR, 'V', 'A', 'U', N, A.data(), N, NULL,
213 NULL, NULL, NULL, 0.0, &m, D.data(), Q.data(), N, supp.data());
214
215 //Determine threshold for regularisation, equal to a prefactor times
216 //N*N*eps times the first (and largest) eigenvalue
217 T tol = 0;
218 for (int i = 0; i < N; ++i)
219 if (std::abs(D(i)) > tol)
220 tol = std::max(tol, std::abs(D(i)));
221 tol *= tolFactor * N * N * std::numeric_limits<T>::epsilon();
222 assert(tol > 0.0);
223
224 //remove eigenvalues below threshhold
225 for (int i = 0; i < N; ++i)
226 if (std::abs(D(i)) < tol)
227 D(i) = 0.0;
228
229 //make A^-1 = QD^-1Q^T
230 blitz::TinyMatrix<T, N, N> QDI(Q);
231 for (int i = 0; i < N; ++i)
232 for (int j = 0; j < N; ++j)
233 if (std::abs(D(j)) > tol)
234 QDI(i, j) /= D(j);
235 else QDI(i, j) = 0.0;
236 lapack::gemm<T>(CblasRowMajor, CblasNoTrans, CblasTrans, N, N, N, 1, QDI.data(), N,
237 Q.data(), N, 0, AInv.data(), N);
238 }

291 ///Nonsymmetric Eigenvalue factorization - for real or complex nonsymmetric NxN matrix: \n
292 ///Schur decomp A = QUQ^T, A^-1 = QU^-1Q^T. Stores A^-1 and eigenvalues.
293 ///U upper triangular with eigenvalues on diagonal. \n
294 ///Regularizing: removes near zero eigen values.
295 ///May be used for inverting block diagonals in SOR. \n
296 ///Note: there may be imaginary eigenvalues, which gives a block triangular U (real Schur form).
297 ///Block sizes are 1 if eig real, 2 if imaginary. For real matrix, blocks are complex conjugate pairs.
298 template<typename T, int N>
299 struct NonSymEigenFactor
300 {
301 using S = lapack::ScalarType_t<T>;
302 blitz::TinyMatrix<T, N, N> U, AInv;
303 blitz::TinyVector<S, N> WR, WI, WM; //real and imaginary parts of evals + magnitude
304 S tol, tolFactor;
305 bool imagE; //imaginary eigenvalues
306
307 NonSymEigenFactor() {}
308
309 NonSymEigenFactor(const blitz::TinyMatrix <T, N, N>& A, S tolFactor = 1.0) :
310 U{ A }, tolFactor{ tolFactor }
311 {
312 //eigen decomp
313 blitz::TinyVector<T, N - 1> tau;
314 lapack::gehrd<T>(LAPACK_ROW_MAJOR, N, 1, N, U.data(), N, tau.data()); //reduce to upper Hessenburg
315 auto Q = U;
316 if constexpr (lapack::is_complex_v<T>)
317 {
318 lapack::unghr<T>(LAPACK_ROW_MAJOR, N, 1, N, Q.data(), N, tau.data()); //generate Q
319 blitz::TinyVector<T, N> W;
320 lapack::hseqr<T>(LAPACK_ROW_MAJOR, 'S', 'V', N, 1, N, U.data(), N, //find eigenvalues
321 W.data(), Q.data(), N);
322 for (int i = 0; i < N; ++i)
323 {
324 WR(i) = W(i).real(); WI(i) = W(i).imag();
325 }
326 }
327 else
328 {
329 lapack::orghr<T>(LAPACK_ROW_MAJOR, N, 1, N, Q.data(), N, tau.data()); //generate Q
330 lapack::hseqr<T>(LAPACK_ROW_MAJOR, 'S', 'V', N, 1, N, U.data(), N, //find eigenvalues
331 WR.data(), WI.data(), Q.data(), N);
332 }
333
334 //magnitude of eigenvalues
335 for (int i = 0; i < N; ++i)
336 WM(i) = std::sqrt(WR(i) * WR(i) + WI(i) * WI(i));
337
338 //Determine threshold for regularisation, equal to a prefactor times
339 //N*N*eps times the first (and largest) eigenvalue
340 tol = 0;
341 for (int i = 0; i < N; ++i)
342 if (WM(i) > tol)
343 tol = std::max(tol, WM(i));
344 tol *= tolFactor * N * N * std::numeric_limits<S>::epsilon();
345 assert(tol > 0.0);
346
347 //check WI all zeros
348 imagE = false;
349 for (const auto& eigI : WI)
350 if (std::abs(eigI) > tol)
351 imagE = true;
352
353 //remove eigenvalues below threshhold
354 for (int i = 0; i < N; ++i)
355 if (std::abs(U(i, i)) < tol)
356 U(i, i) = 0.0;
357
358 //make A^-1 = Q U^-1 Q^T
359 auto ph = transpose(Q);
360 for (int j = 0; j < N; ++j)
361 {
362 blitz::TinyVector<T, N> col;
363 for (int i = 0; i < N; ++i)
364 col(i) = ph(i, j);
365 UInv(col);
366 for (int i = 0; i < N; ++i)
367 ph(i, j) = col(i);
368 }
369 lapack::gemm<T>(CblasRowMajor, CblasNoTrans, CblasNoTrans, N, N, N, 1, Q.data(), N, ph.data(), N, 0, AInv.data(), N);
370 }

372 ///Apply U^-1 via backwards substitution.
373 ///If there are imaginary eigenvalues, use a block backwards substution.
374 void UInv(blitz::TinyVector<T, N>& y) const
375 {
376 for (int i = N - 1; i >= 0; --i)
377 {
378 //determine block size(ie is there a nonzero entry on left side of diagonal?)
379 int blkSize;
380 if (i != 0 && imagE) blkSize = std::abs(U(i, i - 1)) > tol ? 2 : 1;
381 else blkSize = 1;
382
383 if (blkSize == 1)
384 {
385 for (int j = i + 1; j < N; ++j)
386 y(i) -= U(i, j) * y(j);
387 if (U(i, i) == 0.0)
388 y(i) = 0.0;
389 else
390 y(i) /= U(i, i);
391 }
392 else
393 {
394 //block regularize
395
396 //block back subst
397 blitz::TinyVector<T, 2> yii; yii(0) = y(i - 1); yii(1) = y(i);
398 for (int k = 0; k < 2; ++k)
399 for (int j = i + 1; j < N; ++j)
400 yii(k) -= U(i - 1 + k, j) * y(j);
401
402 //apply block Uii^-1
403 T a = U(i - 1, i - 1); T b = U(i - 1, i); T c = U(i, i - 1); T d = U(i, i);
404 blitz::TinyMatrix<T, 2, 2> UiiInv; UiiInv(0, 0) = d; UiiInv(0, 1) = -b;
405 UiiInv(1, 0) = -c; UiiInv(1, 1) = a;
406 UiiInv /= (a * d - b * c);
407 y(i - 1) = UiiInv(0, 0) * yii(0) + UiiInv(0, 1) * yii(1);
408 y(i) = UiiInv(1, 0) * yii(0) + UiiInv(1, 1) * yii(1);
409 --i; //skip past top row of block
410 }
411 }
412 }

588 ///Symmetric eigenvalue factorization - for real, symmetric 2D blitz::Arrays \n
589 ///A = QDQ^T, A^-1 = QD^-1Q^T. \n
590 ///Regularizing: removes near zero eigen values.
591 ///Used for inverting diagonals in SOR, stores A^-1 and D.
592 template<typename T> requires (!is_complex_v<T>)
593 struct SymEigenFactor_Arr
594 {
595 blitz::Array<T, 1> D;
596 blitz::Array<T, 2> AInv;
597 int N;
598 T tolFactor;
599
600 SymEigenFactor_Arr() {}
601
602 SymEigenFactor_Arr(const blitz::Array<T, 2>& A, T tolFactor = 1.0) :
603 tolFactor{ tolFactor }
604 {
605 blitz::Array<T, 2> ACopy{ A.copy() };
606 assert(A.extent(0) == A.extent(1));
607 N = A.extent(0);
608 D.resize(N);
609 AInv.resize(N);
610
611 //eigen decomp
612 lapack_int m;
613 blitz::Array<lapack_int, 1> supp(2 * N);
614 blitz::Array<T, 2> Q(N);
615 lapack::syevr<T>(LAPACK_ROW_MAJOR, 'V', 'A', 'U', N, ACopy.data(), N, NULL,
616 NULL, NULL, NULL, 0.0, &m, D.data(), Q.data(), N, supp.data());
617
618 //Determine threshold for regularisation, equal to a prefactor times
619 //N*N*eps times the first (and largest) eigenvalue
620 T tol = 0;
621 for (int i = 0; i < N; ++i)
622 if (std::abs(D(i)) > tol)
623 tol = std::max(tol, std::abs(D(i)));
624 tol *= tolFactor * N * N * std::numeric_limits<T>::epsilon();
625 assert(tol > 0.0);
626
627 //remove eigenvalues below threshhold
628 for (int i = 0; i < N; ++i)
629 if (std::abs(D(i)) < tol)
630 D(i) = 0.0;
631
632 //make A^-1 = QD^-1Q^T
633 blitz::Array<T, 2> QDI{ Q.copy() };
634 for (int i = 0; i < N; ++i)
635 for (int j = 0; j < N; ++j)
636 if (std::abs(D(j)) > tol)
637 QDI(i, j) /= D(j);
638 else QDI(i, j) = 0.0;
639 lapack::gemm<T>(CblasRowMajor, CblasNoTrans, CblasTrans, N, N, N, 1, QDI.data(), N,
640 Q.data(), N, 0, AInv.data(), N);
641 }

702 ///Same as NonSymEigenFactor, but for 2D blitz::Arrays. \n
703 ///Eigen decomposition of nonsymmetric NxN matrix, real or complex: \n
704 ///Schur decomp A = QUQ^T, A^-1 = QU^-1Q^T, U upper triangular with eigs on diag. \n
705 ///Regularizing: removes near zero eigen values
706 ///Used for robust dense solves, stores A^-1 and eigenvalues
707 ///Note: there may be imaginary eigenvalues, which gives a block triangular U (real Schur form).
708 ///Block sizes are 1 if eig real, 2 if imaginary. For real matrix, blocks are complex conjugate pairs
709 template<typename T>
710 struct NonSymEigenFactor_Arr
711 {
712 using S = lapack::ScalarType_t<T>;
713 blitz::Array<T, 2> U, AInv;
714 blitz::Array<S, 1> WR, WI, WM; ///< Real and imaginary parts of evals + magnitude
715 S tol, tolFactor;
716 int N; ///Size of matrix, must be square.
717 bool imagE; ///Are there imaginary eigenvalues?
718
719 NonSymEigenFactor_Arr() {}
720
721 NonSymEigenFactor_Arr(const blitz::Array<T, 2>& A, S tolFactor = 1.0) :
722 U{ A.copy() }, tolFactor{ tolFactor }
723 {
724 assert(A.extent(0) == A.extent(1));
725 N = A.extent(0);
726 WR.resize(N); WI.resize(N); WM.resize(N);
727
728 //eigen decomp
729 blitz::Array<T, 1> tau(N - 1);
730 lapack::gehrd<T>(LAPACK_ROW_MAJOR, N, 1, N, U.data(), N, tau.data());
731 auto Q{ U.copy() };
732 if constexpr (lapack::is_complex_v<T>)
733 {
734 lapack::unghr<T>(LAPACK_ROW_MAJOR, N, 1, N, Q.data(), N, tau.data()); //generate Q
735 blitz::Array<T, 1> W(N);
736 lapack::hseqr<T>(LAPACK_ROW_MAJOR, 'S', 'V', N, 1, N, U.data(), N, //find eigenvalues
737 W.data(), Q.data(), N);
738 for (int i = 0; i < N; ++i)
739 {
740 WR(i) = W(i).real(); WI(i) = W(i).imag();
741 }
742 }
743 else
744 {
745 lapack::orghr<T>(LAPACK_ROW_MAJOR, N, 1, N, Q.data(), N, tau.data()); //generate Q
746 lapack::hseqr<T>(LAPACK_ROW_MAJOR, 'S', 'V', N, 1, N, U.data(), N, //find eigenvalues
747 WR.data(), WI.data(), Q.data(), N);
748 }
749
750 //magnitude of eigenvalues
751 for (int i = 0; i < N; ++i)
752 WM(i) = std::sqrt(WR(i) * WR(i) + WI(i) * WI(i));
753
754 //Determine threshold for regularisation, equal to a prefactor times
755 //N*N*eps times the first (and largest) eigenvalue
756 tol = 0;
757 for (int i = 0; i < N; ++i)
758 if (WM(i) > tol)
759 tol = std::max(tol, WM(i));
760 tol *= tolFactor * N * N * std::numeric_limits<S>::epsilon();
761 assert(tol > 0.0);
762
763 //check WI all zeros
764 imagE = false;
765 for (const auto& eigI : WI)
766 if (std::abs(eigI) > tol)
767 imagE = true;
768
769 //remove eigenvalues below threshhold
770 for (int i = 0; i < N; ++i)
771 if (std::abs(U(i, i)) < tol)
772 U(i, i) = 0.0;
773
774 //make A^-1 = Q U^-1 Q^T
775 auto ph = transpose(Q);
776 for (int j = 0; j < N; ++j)
777 {
778 blitz::Array<T, 1> col(N);
779 for (int i = 0; i < N; ++i)
780 col(i) = ph(i, j);
781 UInv(col);
782 for (int i = 0; i < N; ++i)
783 ph(i, j) = col(i);
784 }
785 AInv.resize(N);
786 lapack::gemm<T>(CblasRowMajor, CblasNoTrans, CblasNoTrans, N, N, N, 1, Q.data(), N, ph.data(), N, 0, AInv.data(), N);
787 }

789 bool arr_size_check(const blitz::Array<T, 1>& x) const { return x.extent(0) == N; }
790 bool arr_size_check(const blitz::Array<T, 2>& A) const { return A.extent(0) == N && A.extent(1) == N; }
791
792 //apply U^-1 via backwards substitution
793 //if there are imaginary eigenvalues, use a block backwards substution
794 void UInv(blitz::Array<T, 1>& y) const
795 {
796 assert(y.extent(0) == N);
797 for (int i = N - 1; i >= 0; --i)
798 {
799 //determine block size(ie is there a nonzero entry on left side of diagonal?)
800 int blkSize;
801 if (i != 0 && imagE) blkSize = std::abs(U(i, i - 1)) > tol ? 2 : 1;
802 else blkSize = 1;
803
804 if (blkSize == 1)
805 {
806 for (int j = i + 1; j < N; ++j)
807 y(i) -= U(i, j) * y(j);
808 if (U(i, i) == 0.0)
809 y(i) = 0.0;
810 else
811 y(i) /= U(i, i);
812 }
813 else
814 {
815 //block regularize
816
817 //block back subst
818 blitz::TinyVector<T, 2> yii; yii(0) = y(i - 1); yii(1) = y(i);
819 for (int k = 0; k < 2; ++k)
820 for (int j = i + 1; j < N; ++j)
821 yii(k) -= U(i - 1 + k, j) * y(j);
822
823 //apply block Uii^-1
824 T a = U(i - 1, i - 1); T b = U(i - 1, i); T c = U(i, i - 1); T d = U(i, i);
825 blitz::TinyMatrix<T, 2, 2> UiiInv; UiiInv(0, 0) = d; UiiInv(0, 1) = -b;
826 UiiInv(1, 0) = -c; UiiInv(1, 1) = a;
827 UiiInv /= (a * d - b * c);
828 y(i - 1) = UiiInv(0, 0) * yii(0) + UiiInv(0, 1) * yii(1);
829 y(i) = UiiInv(1, 0) * yii(0) + UiiInv(1, 1) * yii(1);
830 --i; //skip past top row of block
831 }
832 }
833 }

904 ///Calculate the eigenvalues of a 2D square blitz::Array.
905 ///Not used for solves/inversion, eigenvalue calculation only.
906 ///Used in Arnodi Ritz value calculations.
907 ///There is an option to calculate the harmonic Ritz values of an upper hessenburg
908 /// matrix H, must include the lower subdiagonal h_mn, see deflated_restart.
909 template<typename T>
910 struct Eigenvalues_Arr
911 {
912 using S = lapack::ScalarType_t<T>;
913 blitz::Array<S, 1> WR, WI, WM; ///< Real and imaginary parts of eigvals + magnitude
914 S tol, tolFactor;
915 int N;
916 bool harmonic_Ritz;
917
918 Eigenvalues_Arr() {}
919 Eigenvalues_Arr(const blitz::Array<T, 2>& A, bool harmonic_Ritz = false, T h_mn = 0.0,
920 S tolFactor = 1.0) : tolFactor{ tolFactor }, harmonic_Ritz{ harmonic_Ritz }
921 {
922 assert(A.extent(0) == A.extent(1)); //square matrix
923 N = A.extent(0);
924 WR.resize(N); WI.resize(N); WM.resize(N);
925 lapack_int sdim;
926 blitz::Array<T, 2> vs(N);
927 auto ACopy{ A.copy() };
928
929 ///If calculating harmonic Ritz values,
930 ///Compute H_tilde = H_n + h_{m,n}^2 * H_n^{-T} * e_n * e_n^T.
931 ///The eigenvalues of H_tilde are the harmonic Ritz values. See deflated_restart.
932 ///H_n is the nxn square portion of H (i.e., A in this construction).
933 ///h_mn the bottom right sub-diagonal element of H.
934 ///H_n^{-T} is the inverse of the transpose of H_n.
935 ///e_n is the n-th standard basis vector.
936 if (harmonic_Ritz)
937 {
938 //Solve H_n^T * w = e_n for w (compute H_n^{-T} * e_n), via LU factorization.
939 blitz::Array<T, 1> w(N); w = T(0);
940 w(N - 1) = T(1); // e_n
941 {
942 LUFactor_Arr<T> H_nT_LU(transpose(A));
943 H_nT_LU.invmul(w);
944 }
945
946 //Form H_tilde = H_n + |h_sub|^2 * w * e_n^T
947 //The second rhs term only modifies the last column of H_n
948 //H_tilde is ACopy in this struct.
949 T scale;
950 if constexpr (is_complex_v<T>) scale = std::conj(h_mn) * h_mn;
951 else scale = h_mn * h_mn;
952 for (int i = 0; i < N; ++i)
953 ACopy(i, N - 1) += scale * w(i);
954 }
955
956 //calculate eigenvalues of ACopy
957 if constexpr (lapack::is_complex_v<T>)
958 {
959 blitz::Array<T, 1> W(N);
960 lapack::gees<T>(LAPACK_ROW_MAJOR, 'N', 'N', nullptr, N, ACopy.data(), N, &sdim,
961 W.data(), vs.data(), N);
962 for (int i = 0; i < N; ++i)
963 {
964 WR(i) = W(i).real(); WI(i) = W(i).imag();
965 }
966 }
967 else
968 lapack::gees<T>(LAPACK_ROW_MAJOR, 'N', 'N', nullptr, N, ACopy.data(), N, &sdim,
969 WR.data(), WI.data(), vs.data(), N);
970
971 //magnitude of eigenvalues
972 for (int i = 0; i < N; ++i)
973 WM(i) = std::sqrt(WR(i) * WR(i) + WI(i) * WI(i));
974 }

1054 ///Solve AX = B for X. A,X,B small, 1x1, 2x2, and 3x3 matrices.
1055 ///if right_solve, solve XA = B.
1056 template<typename T, int N> requires (N <= 3)
1057 blitz::TinyMatrix<T, N, N> small_solve(const blitz::TinyMatrix<T, N, N>& A,
1058 const blitz::TinyMatrix<T, N, N>& B, bool right_solve)
1059 {
1060 if constexpr (N == 1) return blitz::TinyMatrix<T, N, N>(B(0, 0) / A(0, 0));
1061 else if constexpr (N == 2)
1062 {

1064 // [c d] [-c a]
