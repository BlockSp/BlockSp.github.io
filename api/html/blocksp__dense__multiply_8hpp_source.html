23 constexpr int MATVEC_BLAS_THRESH{ 8 }; ///< The threshold between using naive nested loops (nnl)
24 constexpr int MATMAT_BLAS_THRESH{ 8 }; /// vs using Blas for dense matVec and matMat.
25 /// Testing indicates nnl is, in general, faster for small matrices below
26 /// ~6x6-8x8 due to optimized loops over compile time constants.

28 ///Naive nested loop TinyMatrix-TinyVector product, use for small M, N.
29 template<typename T, int M, int N>
30 blitz::TinyVector<T, M> matVec_nnl(const blitz::TinyMatrix<T, M, N>& mat,
31 const blitz::TinyVector<T, N>& vec)
32 {
33 if constexpr (M == 1 && N == 1) return blitz::TinyVector<T, M>(mat(0, 0) * vec(0));
34 if constexpr (M == 1 && N == 2) return blitz::TinyVector<T, M>(mat(0, 0) * vec(0) + mat(0, 1) * vec(1));
35 if constexpr (M == 2 && N == 1) return blitz::TinyVector<T, M>(mat(0, 0) * vec(0), mat(1, 0) * vec(0));
36 if constexpr (M == 2 && N == 2)
37 return blitz::TinyVector<T, M>(mat(0, 0) * vec(0) + mat(0, 1) * vec(1), mat(1, 0) * vec(0) + mat(1, 1) * vec(1));
38
39 blitz::TinyVector<T, M> mv;
40 for (int im = 0; im < M; ++im)
41 {
42 T val = 0;
43 for (int in = 0; in < N; ++in)
44 val += mat(im, in) * vec(in);
45 mv(im) = val;
46 }
47 return mv;
48 }

50 ///TinyMatrix-TinyVector product.
51 ///Using nested loops for small products, Blas for larger.
52 ///For float, double, std::complex<float>, std::complex<double>
53 ///For other types, user must provide implementation.
54 template<typename T, int M, int N>
55 blitz::TinyVector<T, M> matVec(const blitz::TinyMatrix<T, M, N>& mat,
56 const blitz::TinyVector<T, N>& vec)
57 {
58 //compute small mat-vecs first
59 if constexpr (M <= MATVEC_BLAS_THRESH && N <= MATVEC_BLAS_THRESH) //use naive nested loop (nnl)
60 return matVec_nnl(mat, vec);
61 else
62 {
63 //Use Blas for larger matrices
64 blitz::TinyVector<T, M> mv;
65 lapack::gemv<T>(CblasRowMajor, CblasNoTrans, M, N, 1, mat.data(), N, vec.data(), 1, 0, mv.data(), 1);
66 return mv;
67 }
68 }

79 ///Transpose matVec, naive nested loop (nnl). Required due to templated matrix sizes
80 ///NOTE: the conjugate transpose is used for complex types.
81 template<typename T, int M, int N>
82 blitz::TinyVector<T, N> matVec_nnl_transpose(const blitz::TinyMatrix<T, M, N>& mat,
83 const blitz::TinyVector<T, M>& vec)
84 {
85 if constexpr (M == 1 && N == 1) return blitz::TinyVector<T, N>(mat(0, 0) * vec(0));
86 blitz::TinyVector<T, N> mv; mv = 0.0;
87 for (int im = 0; im < M; ++im)
88 for (int in = 0; in < N; ++in)
89 if constexpr (is_complex_v<T>) mv(in) += std::conj(mat(im, in)) * vec(im);
90 else mv(in) += mat(im, in) * vec(im);
91 return mv;
92 }

127 ///TinyMatrix-TinyMatrix product
128 ///between MxK and KxN matrices : resulting MxN
129 template<typename T, int M, int N, int K>
130 blitz::TinyMatrix<T, M, N> matMat(const blitz::TinyMatrix<T, M, K>& mat1, const blitz::TinyMatrix<T, K, N>& mat2)
131 {
132 //small matrix-matrix products first
133 if constexpr (M == 1 && K == 1 && N == 1) return blitz::TinyMatrix<T, M, N>(mat1(0, 0) * mat2(0, 0));
134 else if constexpr (M <= MATMAT_BLAS_THRESH && K <= MATMAT_BLAS_THRESH &&
135 N <= MATMAT_BLAS_THRESH) //faster to use nested loops
136 {
137 blitz::TinyMatrix<T, M, N> mm(0.0);
138 for (int im = 0; im < M; ++im)
139 for (int ik = 0; ik < K; ++ik)
140 for (int in = 0; in < N; ++in)
141 mm(im, in) += mat1(im, ik) * mat2(ik, in);
142 return mm;
143 }
144 else
145 {
146 //Use Blas for larger matrix-matrix products
147 blitz::TinyMatrix<T, M, N> mm;
148 lapack::gemm<T>(CblasRowMajor, CblasNoTrans, CblasNoTrans, M, N, K, 1, mat1.data(), K,
149 mat2.data(), N, 0, mm.data(), N);
150 return mm;
151 }
152 }

154 ///Transpose matMat - first matrix only: A^T B - results in K x N matrix
155 ///NOTE: the conjugate transpose is used for complex types.
156 template<typename T, int M, int N, int K>
157 blitz::TinyMatrix<T, K, N> matMat_transpose(const blitz::TinyMatrix<T, M, K>& mat1, const blitz::TinyMatrix<T, M, N>& mat2)
158 {
159 //small matrix-matrix products first
160 if constexpr (M <= MATMAT_BLAS_THRESH && K <= MATMAT_BLAS_THRESH &&
161 N <= MATMAT_BLAS_THRESH) //faster to use nested loops
162 {
163 blitz::TinyMatrix<T, K, N> mm(0.0);
164 for (int im = 0; im < M; ++im)
165 for (int ik = 0; ik < K; ++ik)
166 for (int in = 0; in < N; ++in)
167 if constexpr (is_complex_v<T>) mm(ik, in) += std::conj(mat1(im, ik)) * mat2(im, in);
168 else mm(ik, in) += mat1(im, ik) * mat2(im, in);
169 return mm;
170 }
171 else
172 {
173 //Use Blas for larger matrix-matrix products
174 blitz::TinyMatrix<T, K, N> mm;
175 if constexpr (is_complex_v<T>)

177 mat2.data(), N, 0, mm.data(), N);
178 else lapack::gemm<T>(CblasRowMajor, CblasTrans, CblasNoTrans, K, N, M, 1, mat1.data(), K,
179 mat2.data(), N, 0, mm.data(), N);
180 return mm;
181 }

206 ///Transpose of dense matrix, floating point numbers, via mkl_?omatcopy
207 ///NOTE: the conjugate transpose is used for complex types.
208 template<typename T, int M, int N>
209 blitz::TinyMatrix<T, N, M> transpose(const blitz::TinyMatrix<T, M, N>& mat)
210 {
211 blitz::TinyMatrix<T, N, M> tran;
212 if constexpr (lapack::is_complex_v<T>)
213 lapack::omatcopy<T>('r', 'c', M, N, T(1.0), mat.data(), N, tran.data(), M); //conjugate transpose
214 else lapack::omatcopy<T>('r', 't', M, N, T(1.0), mat.data(), N, tran.data(), M); //transpose
215 return tran;
216 }
