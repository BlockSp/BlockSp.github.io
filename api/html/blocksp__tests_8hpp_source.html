592 ///Read in a matrix-market matrix file (.mtx) and convert it to a BlockSp matrix.
593 ///If symmetric, only half is in file, must make entire matrix.
594 ///If base_zero, the matrix indexing in the .mtx file starts from zero
595 template<typename T, int B>
596 BlockSp<T, B> mtxMatrix_to_BlockSp(const std::string& mat_loc,
597 unsigned int start_line, bool symmetric, bool base_zero, bool printMatSize)
598 {
599 std::ifstream f(mat_loc, std::ios::in);
600
601 //Matrix market files are formatted in following manner
602 //1) Name of matrix / matrix details
603 //2) Number of rows, number of columns, number of entries/nonzeros
604 //3+) Values of matrix in matrix market format
605 //Matrix market format: row index, col index, value
606 GotoLine(f, start_line); //user must ensure starting line is matrix size line (2)
607 std::string size; std::getline(f, size);
608 std::stringstream ss(size);
609 int M, N, nnz; ss >> M; ss >> N; ss >> nnz;
610 if (printMatSize)
611 {
612 std::cout << M << ' ' << N << ' ' << nnz;
613 if (symmetric) std::cout << " Symmetric";
614 std::cout << std::endl;
615 }
616
617 //Note the choice of block size B for the BlockSp matrix must divide
618 //mtx's number of rows M and number of columns N
619 assert(M % B == 0 && N % B == 0);
620 int lm = M / B; //number of rows in BlockSp matrix
621 int ln = N / B; //number of cols in BlockSp matrix
622 BlockSp<T, B> A(lm, ln);
623
624 //Read values and fill BlockSp matrix.
625 for (int innz = 0; innz < nnz; ++innz)
626 {
627 std::string entry; std::getline(f, entry);
628 int i, j; T val;
629 std::stringstream ss2(entry);
630 ss2 >> i; ss2 >> j; //write matrix index and values.
631 i -= 1; j -= 1; //change to base 0 indexing.
632 if constexpr (is_complex_v<T>) //also read imaginary value.
633 {
634 ScalarType_t<T> valReal, valImg; ss2 >> valReal; ss2 >> valImg;
635 val.real(valReal); val.imag(valImg);
636 util::setFromGlobal(A, i, j, val);
637 if (symmetric && i != j) //if symmetric, fill (j,i) with complex conjugate.
638 util::setFromGlobal(A, j, i, std::conj(val));
639 }
640 else
641 {
642 ss2 >> val;
643 util::setFromGlobal(A, i, j, val);
644 if (symmetric && i != j) //if symmetric, fill (j,i)
645 util::setFromGlobal(A, j, i, val);
646 }
647 }
648 return A;
649 }

651 ///Read in a matrix-market vector file (.mtx) and convert it to a solArray.
652 template<typename T, int B, int D>
653 solArray<T, B, D> mtxArray_to_solArray(const std::string& vec_loc,
654 unsigned int start_line, bool printVecSize)
655 {
656 std::ifstream f(vec_loc, std::ios::in);
657
658 //matrix market files are formatted in following manner
659 //1) name of vector
660 //2) number of rows, number of cols (1)
661 //3+) values of vector
662 GotoLine(f, start_line); //user must ensure starting line is vector size line
663 std::string size; std::getline(f, size);
664 std::stringstream ss(size);
665 int M, N; ss >> M; ss >> N;
666 if (printVecSize) std::cout << M << ' ' << N << std::endl;
667
668 //Note the choice of block size B and dimension D for the solArray must divide
669 //mtx's number of rows M, N must = 1
670 int constexpr BD = B * D;
671 assert(M % BD == 0 && N == 1);
672 int l = M / BD; //number of rows in BlockSp matrix
673 solArray<T, B, D> x(l);
674
675 //convert mtx array index to sol index and fill it's value
676 auto fill_solArray = [&](int i, T val)
677 {
678 int im = i / BD; //find the index of the solArray
679 int i_BD = i % B; //find the index within the local blocks
680 int dim = i_BD / B;
681 int ib = i_BD % B;
682 x(im, dim)(ib) = val; //fill value
683 };
684
685 for (int im = 0; im < M; ++im)
686 {
687 std::string entry; std::getline(f, entry);
688 std::stringstream ss2(entry);
689 T val;
690 if constexpr (is_complex_v<T>) //also read imaginary value.
691 {
692 long double valReal, valImg; ss2 >> valReal; ss2 >> valImg;
693 val.real(valReal); val.imag(valImg);
694 }
695 else ss2 >> val;
696 fill_solArray(im, val);
697 }
698 return x;
699 }
