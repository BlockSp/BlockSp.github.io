30 ///Transpose a BlockSp matrix.
31 template<typename T, int M>
32 BlockSp<T, M> blkSpTranspose(const BlockSp<T, M>& other)
33 {
34 BlockSp<T, M> tran(other.n(), other.m());
35 for (int im = 0; im < other.m(); ++im)
36 for (int in = 0; in < other.colSize(im); ++in)
37 {
38 int colInd = other.colInd(im, in);
39 tran(colInd, im) = dense::transpose<T, M, M>(other(im, colInd));
40 }
41 return tran;
42 }

49 ///Block sparse matrix-matrix multiplication (bsmm).
50 ///C = A * B or C = A^T * B. bool Atran transposes A.
51 template<typename T, int M>
52 BlockSp<T, M> blkSpMatMat(const BlockSp<T, M>& A, const BlockSp<T, M>& B, bool Atran = false)
53 {
54 if (!Atran) assert(A.n() == B.m());
55 else assert(A.m() == B.m());
56
57 BlockSp<T, M> C;
58 int m_m = A.m();
59 int m_n = A.n();
60 if (!Atran) C = BlockSp<T, M>(m_m, B.n());
61 else C = BlockSp<T, M>(m_n, B.n());
62
63 if (!Atran)
64 {
65 //pre determine sparsity structure
66 std::vector<int> sparsity(m_m, 0);
67 for (int imA = 0; imA < A.m(); ++imA) //scan each row of A
68 { //scan each column in that row of A, its col number...
69 for (int inA = 0; inA < A.colSize(imA); ++inA)
70 {
71 int colIndA = A.colInd(imA, inA);
72 //...becomes the row number in B, scan each column in that row of B
73 for (int inB = 0; inB < B.colSize(colIndA); ++inB)
74 ++sparsity[imA];
75 }
76 }
77 //reserve space in the BlockSp's vectors, matching the sparsity structure
78 for (int imA = 0; imA < m_m; ++imA)
79 {
80 C.reserve_col(imA, sparsity[imA]);
81 C.reserve_data(imA, sparsity[imA]);
82 }
83 //perform matrix-matrix product
84 //scan each row of A
85 for (int imA = 0; imA < m_m; ++imA)
86 { //scan each column in that row of A, its col number...
87 for (int inA = 0; inA < A.colSize(imA); ++inA)
88 {
89 int colIndA = A.colInd(imA, inA);
90 //...becomes the row number in B, scan each column in that row of B
91 for (int inB = 0; inB < B.colSize(colIndA); ++inB)
92 {
93 int colIndB = B.colInd(colIndA, inB);
94 C(imA, colIndB) += dense::matMat<T, M>(A(imA, colIndA), B(colIndA, colIndB));
95 }
96 }
97 }
98 }
99 else
100 {
101 //pre determine sparsity structure
102 std::vector<int> sparsity(m_n, 0);
103 for (int imA = 0; imA < A.m(); ++imA) //scan each row of A
104 { //scan each column in that row of A, for transpose its row number is now col num which ...
105 for (int inA = 0; inA < A.colSize(imA); ++inA)
106 {
107 int colIndA = A.colInd(imA, inA);
108 //...becomes the row number in B, scan each column in that row of B
109 for (int inB = 0; inB < B.colSize(imA); ++inB)
110 ++sparsity[colIndA];
111 }
112 }
113 //reserve space in the BlockSp's vectors, matching the sparsity structure
114 for (int inA = 0; inA < m_n; ++inA)
115 {
116 C.reserve_col(inA, sparsity[inA]);
117 C.reserve_data(inA, sparsity[inA]);
118 }
119 //perform matrix-matrix product
120 //scan each row of A
121 for (int imA = 0; imA < A.m(); ++imA)
122 { //scan each column in that row of A, for trans its row number is now col num which ...
123 for (int inA = 0; inA < A.colSize(imA); ++inA)
124 {
125 int colIndA = A.colInd(imA, inA);
126 //...becomes the row number in B, scan each column in that row of B
127 for (int inB = 0; inB < B.colSize(imA); ++inB)
128 {
129 int colIndB = B.colInd(imA, inB);
130 C(colIndA, colIndB) += dense::matMat<T, M>(A(imA, colIndA), B(imA, colIndB), true);
131 }
132 }
133 }
134 }
135 return C;
136 }

166 ///Matrix vector product (bsmv) between BxB BlockSp matrix with solArray of size BxD.
167 //Multiply either a single dimension (dim >= 0) or all dimensions (set dim < 0, default).
168 ///A^T if transpose.
169 ///This function also encapsulates scalar cases, D = 1.
170 template<typename T, int B, int D = 1>
171 solArray<T, B, D> blkSpMatVec(const BlockSp<T, B>& A, const solArray<T, B, D>& x,
172 int dim = -1, bool transpose = false)
173 {
174 if (!transpose) assert(A.n() == x.extent(0));
175 else assert(A.m() == x.extent(0));
176 assert(dim < D);
177
178 if (!transpose)
179 {
180 int nRows = A.m();
181 solArray<T, B, D> b{ nRows };
182 for (int im = 0; im < nRows; ++im)
183 {
184 int in = 0;
185 for (const auto& a : A.get_data(im))
186 {
187 int col = A.colInd(im, in);
188 if(dim >= 0) b(im, dim) += dense::matVec<T, B, B>(a, x(col, dim));
189 else for (int dim2 = 0; dim2 < D; ++dim2)
190 b(im, dim2) += dense::matVec<T, B, B>(a, x(col, dim2));
191 ++in;
192 }
193 }
194 return b;
195 }
196 else
197 {
198 solArray<T, B, D> b{ A.n() };
199 for (const auto& index : A.getIndices())
200 if(dim >= 0) b(index(1), dim) += dense::matVec_transpose<T, B, B>(A(index), x(index(0), dim));
201 else for (int dim2 = 0; dim2 < D; ++dim2)
202 b(index(1), dim2) += dense::matVec_transpose<T, B, B>(A(index), x(index(0), dim2));
203 return b;
204 }
205 }
