21 ///Make a vector of inverses of the block diagonal of a BlockSp matrix
22 template<typename Inv, typename T, int B>
23 std::vector<Inv> makeADInv(const BlockSp<T, B>& A)
24 {
25 //Test all diagonal blocks for zero eigenvalues to ensure they are invertable.
26 //Will break program if zero eigenvalue found.
27 if constexpr (test_zero_eigenvalues)
28 {
29 ScalarType_t<T> mxc = 0.0;
30 ScalarType_t<T> mx = 0.0;
31 ScalarType_t<T> mn = std::numeric_limits<ScalarType_t<T>>::max();
32 for (int im = 0; im < A.m(); ++im)
33 {
34 if constexpr (B == 1)
35 {
36 ScalarType_t<T> val = std::abs(A(im, im)(0, 0));
37 mn = std::min(mn, val);
38 mx = std::max(mx, val);
39 }
40 else
41 {
42 dense::NonSymEigenFactor<T, B> ee(A(im, im));
43 mn = std::min(mn, ee.min_mag());
44 mx = std::max(mx, ee.max_mag());
45 mxc = std::max(mxc, ee.cond_number());
46 }
47 if (mn < 1.0e-15)
48 {
49 std::cout << "Zero eigenvalue on diagonal!" << std::endl;
50 std::cout << "Cannot invert block " << im << ", goodbye." << std::endl;
51 std::exit(EXIT_FAILURE);
52 }
53 }
54 std::cout << "min block eval " << mn << std::endl;
55 std::cout << "max block eval " << mx << std::endl;
56 if (B != 1) std::cout << "max block cond# " << mxc << std::endl;
57 }
58
59 std::vector<Inv> AD_Inv(A.m());
60 for (int im = 0; im < A.m(); ++im)
61 AD_Inv[im] = Inv(A(im, im));
62 return AD_Inv;
63 }

132 ///One iteration of SOR in the forward direction.
133 ///Works to solve Ax = b, overwrites initial guess x.
134 ///w = SOR damping parameter.
135 void oneSORForw(const BlockSp<T, B* D>& A, solArray<T, B, D>& x, const solArray<T, B, D>& b, ScalarType_t<T> w) const
136 {
137 assert(A.m() == b.len() && A.n() == x.len());
138
139 //forward SOR iterations
140 for (int im = 0; im < A.m(); ++im)
141 {
142 blitz::TinyVector<T, B* D> ph(0.0);
143 int in = 0;
144 for (const auto& a : A.get_data(im))
145 {
146 int colInd = A.colInd(im, in);
147 if (colInd != im)
148 {
149 if constexpr (D == 1) ph += dense::matVec<T, B* D, B* D>(a, x(colInd));
150 else
151 {
152 auto xLoc{ util::tvConcat<T, B, D>(x(colInd)) };
153 ph += dense::matVec<T, B* D, B* D>(a, xLoc);
154 }
155 }
156 ++in;
157 }
158 if constexpr (D == 1) ph = b(im) - ph;
159 else ph = util::tvConcat<T, B, D>(b(im)) - ph;
160 AD_Inv[im].invmul(ph);
161 if constexpr (D == 1)
162 x(im) += -w * x(im) + w * ph;
163 else for (int dim = 0; dim < D; ++dim)
164 x(im, dim) += -w * x(im, dim) + w * util::tvSubset<T, B, D>(ph, dim);
165 }
166 };

244 ///Approximates Ax = b by factoring A via LU decomposition without fill-in.
245 /// No pivoting.
246 template<typename T, int B, int D = 1>
247 struct ILU0
248 {
249 BlockSp<T, B* D> LU;
250 std::vector<std::unordered_set<int>> rowInd; ///< The nonzero row indices for each column
251 std::vector<dense::LUFactor<T, B* D>> D_Inv; ///< The inverse of the Block diagonal of LU, after factor.
252 int n; ///< The BlockSp matrix size.

254 ILU0() : n{ 0 } {}
255 ILU0(const BlockSp<T, B* D>& A) : LU{ A }
256 {
257 assert(A.m() == A.n());
258 n = A.n();
259 rowInd = A.rowIndicesByCol(); //non-zero row indices of A, sorted by column.
260 factor(LU); //perform ILU0 factoriation
261 D_Inv = makeADInv<dense::LUFactor<T, B* D>>(LU); //LU factor the diagonal blocks
262 LU.shrink_to_fit(); //remove excess vector storage from LU
263 }
264 ILU0& operator= (const ILU0& other)

268 D_Inv = other.D_Inv;

273 //Perform an inplace ILU0 factorization of matrix A
274 void factor(BlockSp<T, B* D>& A)
275 {
276 for (int i = 0; i < n; ++i)
277 {
278 //Factor below diagonal, only for A's non-zero entries.
279 //Note: Gaussian elimination calls for application of the "right" inverse,
280 // i.e, L_ki = A_ki * A^{-1}_ii
281 dense::LUFactor<T, B* D> AD_Inv(A(i, i));
282 for (const auto& k : rowInd[i])
283 if (k > i) AD_Inv.r_invmul(A(k, i));
284
285 //Gaussian eliminate the remaining rows/columns
286 for (const auto& k : rowInd[i]) if (k > i)
287 for (const auto& j : A.colInd(k)) if (j > i)
288 if (rowInd[j].contains(i))
289 A(k, j) -= dense::matMat(A(k, i), A(i, j));
290 }
291 }

359 ///Block Incomplete Cholesky(0) preconditioner.
360 ///Approximates Ax = b by factoring A via Cholesky decomposition LL^T without fill-in.
361 ///No pivoting. Using lower triangular matrices.
362 ///For symmetric positive definite (SPD) linear systems,
363 /// which are the user's responsibility to provide.
364 template<typename T, int B, int D = 1>
365 struct IChol0
366 {
367 BlockSp<T, B* D> L; ///< The IChol0 factorization
368 std::vector<std::unordered_set<int>> rowInd; ///< The nonzero row indices for each column
369 int n; ///< The BlockSp matrix size.

373 ///Construct IChol0 from an SPD matrix A. A is assumed to be full, not triangular.
374 IChol0(const BlockSp<T, B* D>& A) : L{ A.lowTri() } //extract lower triangular of A.
375 {
376 assert(A.m() == A.n());
377 n = A.n();
378 rowInd = A.rowIndicesByCol(); //non-zero row indices of A, sorted by column.
379 factor(L); //create IChol0 factorization L.
380 rowInd = L.rowIndicesByCol(); //update row indices to be lower triangular L.
381 L.shrink_to_fit(); //remove excess vector storage from L
382 }
383 IChol0& operator= (const IChol0& other)
384 {
385 L = other.L;
386 rowInd = other.rowInd;
387 n = other.n;
388 return *this;
389 }

391 //Perform an inplace IChol0 factorization of matrix A_l.
392 //Assumes that A_l is the lower block triangular of an SPD matrix.
393 void factor(BlockSp<T, B* D>& A_l)
394 {
395 int constexpr BD = B * D;
396 for (int i = 0; i < n; ++i)
397 {
398 //Cholesky factor the diagonal L_ii
399 if constexpr (BD == 1) A_l(i, i) = std::sqrt(A_l(i, i)(0, 0));
400 else
401 {
402 lapack::potrf<T>(LAPACK_ROW_MAJOR, 'L', BD, A_l(i, i).data(), BD);
403 for (int ib = 0; ib < BD; ++ib)
404 for (int jb = ib + 1; jb < BD; ++jb)
405 A_l(i, i)(ib, jb) = T(0.0); //set upper triangular of diag block to zero
406 }
407
408 //Factor below diagonal, only for A's non-zero entries.
409 //Note: Gaussian elimination calls for application of the "right" inverse,
410 // i.e, L_ki = A_ki * L^{-T}_ii
411 for (const auto& k : rowInd[i]) if (k > i)
412 {
413 if constexpr (BD == 1) A_l(k, i) /= A_l(i, i)(0, 0);
414 else
415 {
416 //right inverse solved via A^T X^T = Y^T
417 dense::transpose_in_place(A_l(k, i)); //make Y^T
418 Lii_Inv_forward_subst(A_l(i, i), A_l(k, i));
419 dense::transpose_in_place(A_l(k, i)); //get X
420 }
421 }
422
423 //Gaussian eliminate the remaining rows/columns
424 for (const auto& k : rowInd[i]) if (k > i)
425 for (const auto& j : A_l.colInd(k)) if (j > i && j <= k)
426 if (rowInd[j].contains(i))
427 A_l(k, j) -= dense::matMat(A_l(k, i), dense::transpose(A_l(j, i)));
428 }
429 }
