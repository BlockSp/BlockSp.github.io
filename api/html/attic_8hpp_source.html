3 ///Return the minor of a matrix A at i,j
4 ///i.e. the determinant of a sub matrix of A with row i and col j removed
5 ///Warning! Explicit determinant calculations are an O(N!), where ! is factorial.
6 //Use for small N only.
7 template<typename T, int N>
8 T minor(const blitz::TinyMatrix<T, N, N>& A, int i, int j)
9 {
10 if constexpr (N == 1) return A(0, 0);
11 else
12 {
13 //make submatrix
14 blitz::TinyMatrix<T, N - 1, N - 1> ASub;
15 for (int is = 0; is < N; ++is) if (is != i)
16 for (int js = 0; js < N; ++js) if (js != j)
17 {
18 int row_s = is > i ? is - 1 : is; //find row index for ASub
19 int col_s = js > j ? js - 1 : js; //find column index for ASub
20 ASub(row_s, col_s) = A(is, js);
21 }
22 return determinant(ASub);
23 }
24 }

26 //Return the adjunct matrix for matrix A
27 //Adjunct is the transpose of the Cofactor matrix
28 template<typename T, int N>
29 blitz::TinyMatrix<T, N, N> adjunct(const blitz::TinyMatrix<T, N, N>& A)
30 {
31 blitz::TinyMatrix<T, N, N> adj;
32 for (int i = 0; i < N; ++i)
33 for (int j = 0; j < N; ++j)
34 adj(i, j) = std::pow(-1.0, i + j) * minor(A, j, i);
35 return adj;
36 }

38 ///Calculate the explicit inverse of a TinyMatrix A
39 ///Uses the formula A^-1 = 1/det(A) * adj(A).
40 ///Warning! Should only be used for very small N.
41 ///In general, avoid explicit inverses, see blocksp_dense.hpp faster dense solvers.
42 template<typename T, int N>
43 blitz::TinyMatrix<T, N, N> explicit_Inv(const blitz::TinyMatrix<T, N, N>& A)
44 {
45 if constexpr (N == 1) return 1.0 / A(0, 0);
46 else if constexpr (N == 2)
47 {
48 T det = A(0, 0) * A(1, 1) - A(1, 0) * A(0, 1);
49 blitz::TinyMatrix<T, N, N> A_Inv;
50 A_Inv(0, 0) = A(1, 1); A_Inv(1, 1) = A(0, 0);
51 A_Inv(1, 0) = -A(0, 1); A_Inv(0, 1) = -A(1, 0);
52 A_Inv /= det;
53 return A_Inv;
54 }
55 else
56 {
57 auto A_Inv{ adjunct(A) };
58 A_Inv /= determinant(A);
59 return A_Inv;
60 }

327 ///Harmonic Ritz vectors are picked to approximate the smallest eigenvalues/vectors.
328 ///These are the eigenvalues that give pGMRES the most difficulty.
329 ///See "GMRES with Deflated Restarting" by Ronald B. Morgan for more details.
330 ///
331 ///Reasoning: at iteration n, GMRES approximates the residual r_n = p_n(A)r_0,
332 ///where p_n is an n_th degree polynomial and r_0 is the initial residual.
333 ///The polynomial is such that p_n(0) = 1,
334 ///also p_n crosses through zero near eigenvalues. Removing small eigenvalues allows the
335 ///polynomial to remain smaller near other points, and allows GMRES to converge faster.
336 ///Restarted pGMRES with k = 0 will reexplore these small eigenvalues and may stall.
337 ///pGMRES without restart may become intractably large/slow.

1006void deflated_restart_3(blitz::Array<T, 2>& H, std::vector<solArray<T, B, D>>& Qn,

1012 int n = H.extent(1); // n = number of columns
1013 S zero_thresh = 10 * n * n * std::numeric_limits<S>::epsilon();

1015 //std::cout << "deflated_restart: H is " << m << " x " << n << std::endl;
1016
1017 if (k == 0) // pure restart
1018 {
1019 H.resize(0);
1020 Qn.clear();
1021 res.clear();

1023 return;
1024 }
1025
1026 // Limit k to available eigenvalues
1027 if (k > n) k = n;
1028
1029 //////////////////////////////////////////////////////
1030 /// (1) Compute H_tilde = H_n + h_{m,n}^2 * H_n^{-T} * e_n * e_n^T
1031 /// The eigenvalues of H_tilde are the harmonic Ritz values.
1032
1033 // Extract square portion H_n (n x n)
1034 blitz::Array<T, 2> H_n(n, n);
1035 for (int i = 0; i < n; ++i)
1036 for (int j = 0; j < n; ++j)
1037 H_n(i, j) = H(i, j);
1038

1040 T h_sub = H(n, n - 1);

1042 // Solve H_n^T * w = e_n for w (i.e., compute H_n^{-T} * e_n)
1043 blitz::Array<T, 1> w(n);
1044 w = T(0);
1045 w(n - 1) = T(1); // e_n
1046
1047 // Transpose H_n for the solve
1048 blitz::Array<T, 2> H_nT(n, n);
1049 for (int i = 0; i < n; ++i)
1050 for (int j = 0; j < n; ++j)
1051 H_nT(i, j) = H_n(j, i);

1053 // Solve H_n^T * w = e_n using LU factorization
1054 {
1055 dense::LUFactor_Arr<T> H_nT_LU(H_nT);
1056 H_nT_LU.invmul(w);
1057 }
1058
1059 // Form H_tilde = H_n + h_sub^2 * w * e_n^T
1060 // This only modifies the last column of H_n

1063 for (int i = 0; i < n; ++i)
1064 H_tilde(i, n - 1) += scale * w(i);

1082 std::iota(perm.begin(), perm.end(), 0);
1083 std::sort(perm.begin(), perm.end(), [&](int a, int b) {
1084 return std::abs(eig_vals(a)) < std::abs(eig_vals(b));
1085 });

1089 for (int i = 0; i < k; ++i)
1090 select[perm[i]] = 1;

1095 reinterpret_cast<MKL_Complex16*>(H_tilde.data()), n,
1096 reinterpret_cast<MKL_Complex16*>(Z.data()), n,

1099 //lapack::trsen<T>(LAPACK_ROW_MAJOR, 'N', 'V', select.data(), n,
1100 // H_tilde.data(), n, Z.data(), n, eig_vals.data(), &sdim, &s_out, &sep_out);
1101 }
1102 else
1103 {
1104 // Real case: use dgees
1105 blitz::Array<T, 1> wr(n), wi(n); // real and imaginary parts of eigenvalues
1106 lapack::gees<T>(LAPACK_ROW_MAJOR, 'V', 'N', nullptr, n,
1107 H_tilde.data(), n, &sdim, wr.data(), wi.data(), Z.data(), n);
1108
1109 // Sort eigenvalues by magnitude to find k smallest
1110 std::vector<int> perm(n);
1111 std::iota(perm.begin(), perm.end(), 0);
1112 std::sort(perm.begin(), perm.end(), [&](int a, int b) {
1113 S mag_a = std::sqrt(wr(a) * wr(a) + wi(a) * wi(a));
1114 S mag_b = std::sqrt(wr(b) * wr(b) + wi(b) * wi(b));
1115 return mag_a < mag_b;
1116 });
