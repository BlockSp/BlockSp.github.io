83 ///Engine to compare the timing of Eigen's sparse matrix iterative solvers
84 /// with BlockSp's iterative solvers.
85 ///
86 ///Reads in an Eigen_Trip<T> A_trip that constructs sparse matrix A, and
87 /// a pointer to an Eigen::Vector for the exact solution x or right-hand-side b.
88 ///Then solves Ax = b.
89 ///
90 ///The triplet A_Trip and vectors x/b may be contructed from matrix market files .mtx
91 /// via the mtx file converters in this file. From these, the BlockSp structures are created.
92 ///
93 ///If x and b are provided. Ax = b is solved and compared to the exact solution.
94 ///If only x is provided, b = Ax is calculated, then used as input for the solvers.
95 ///If only b is provided, Ax = b is first solved by Eigen's SparseLU direct solver,
96 /// this is used as a reference solution. Note: direct solvers may not be suitable for large matrices.
97 ///If neither x nor b is provided, a random normal solution vector x is calculated
98 /// and rhs b generated.
99 ///
100 ///Tests
101 ///Compares Eigen's BICGSTAB with diagonal and IncompleteLUT preconditioners
102 ///to BlockSp's pGMRES with block SOR and ILU0 preconditioners.
103 ///
104 ///If symmetric positive definite (SPD), we also test both libraries'
105 /// preconditioneed conjugate gradient (pCG).
106 ///For Eigen, we test diagonal and IncompleteCholesky preconditioners.
107 ///For BlockSp, we test symmetric SOR and IChol0 preconditioners.
108 ///
109 ///If noPre, test solvers with no preconditioner.
110 ///If diag, test BlockSp solvers with a block diagonal preconditioner.
111 ///If robust, test BlockSp's ez::sor_pGMRES_robust.
112 ///If dr, run deflated restart versions for BlockSp's pGMRES.
113 template<typename T, int B, int D = 1>
114 void eigen_compare(const Eigen_Trip<T>& A_trip, const Eigen::Vector<T, Eigen::Dynamic>* x_ptr,

461 ///Compare the performance of BlockSp to Eigen
462 ///Solve a random symmetric BlockSp matrix with a random solArray(a multi-dimensional vector).
463 ///The matrix is outlined similarly to a heat operator: ASym = (I - d A^T A).
464 ///Can use data type T = float, double, std::complex<float>, or std::complex<double>.
465 template<typename T>
466 void block_pseudo_heat()
467 {
468 //Try changing D, B, l, nCol, and delta.
469 int constexpr D = 2; //vector Dimension //D and B are compile time constants!
470 int constexpr B = 16; //vector Blocksize
471 int constexpr BD = B * D;
472 using S = BlockSp::ScalarType_t<T>; //float or double, derived from T.
473 int l = 256; //vector/BlockSp matrix length
474 int eig_l = l * B * D;
475
476 //make random BlockSp matrix A and symmetric positive definite matrix ASym = I - delta*A^T*A
477 int nCol = 4; //num of nonzero columns's per row of A. ASym will have ~2*nCol nonzero columns per row.
478 BlockSp::BlockSp<T, BD> A{ BlockSp::util::rand_BlockSp<T, BD>(l, l, nCol) };
479 S delta = 2.0e-4; //the value of delta may be tuned to stress test the solvers.
480 auto ASym{ BlockSp::bs_id<T, BD>(l) - delta * BlockSp::bsmm(A, A, true) }; //ASym = I - delta*A^T*A
481 std::cout << "block_pseudo_heat: data type " << typeid(T).name() << std::endl;
482
483 //convert BlockSp to Eigen_Trip
484 Eigen_Trip<T> trip;
485 trip.M = eig_l; trip.N = eig_l;
486
487 //Convert BlockSp index to triplet index and fill it's value
488 auto fill_trip = [&](int i, int j, const blitz::TinyMatrix<T, BD, BD>& blk)
489 {
490 int i_start = i * BD; //starting index for triplets
491 int j_start = j * BD;
492 for (int i_blk = 0; i_blk < BD; ++i_blk)
493 for (int j_blk = 0; j_blk < BD; ++j_blk)
494 {
495 int i_trip = i_start + i_blk;
496 int j_trip = j_start + j_blk;
497 trip.push_back(Eigen::Triplet<T>(i_trip, j_trip, blk(i_blk, j_blk)));
498 }
499 };
500 for (int im = 0; im < l; ++im)
501 for (const auto& icol : ASym.colInd(im))
502 fill_trip(im, icol, ASym(im, icol));
503

521 ///Eigen sparse matrix test.
522 ///Includes the functions insertCoefficient and buildProblem.
523 ///Builds a straightforward sparse 2D Poisson finite difference problem.
524 ///Resulting matrix is symmetric positive definite (SPD).
525 void insertCoefficient(int id, int i, int j, double w, std::vector<Eigen::Triplet<double>>& coeffs,
526 Eigen::VectorXd& b, const Eigen::VectorXd& boundary)
527 {
528 int n = int(boundary.size());
529 int id1 = i + j * n;
530
531 if (i == -1 || i == n)
532 b(id) -= w * boundary(j); // constrained coefficient
533 else if (j == -1 || j == n)
534 b(id) -= w * boundary(i); // constrained coefficient
535 else
536 coeffs.push_back(Eigen::Triplet<double>(id, id1, w)); // unknown coefficient
537 }
538 void buildProblem(std::vector<Eigen::Triplet<double>>& coefficients, Eigen::VectorXd& b, int n)
539 {
540 b.setZero();
541 Eigen::ArrayXd boundary = Eigen::ArrayXd::LinSpaced(n, 0, EIGEN_PI).sin().pow(2);
542 for (int j = 0; j < n; ++j) {
543 for (int i = 0; i < n; ++i) {
544 int id = i + j * n;
545 insertCoefficient(id, i - 1, j, -1, coefficients, b, boundary);
546 insertCoefficient(id, i + 1, j, -1, coefficients, b, boundary);
547 insertCoefficient(id, i, j - 1, -1, coefficients, b, boundary);
548 insertCoefficient(id, i, j + 1, -1, coefficients, b, boundary);
549 insertCoefficient(id, i, j, 4, coefficients, b, boundary);
550 }
551 }
552 }

727 ///Build a BlockSp matrix from an Eigen_Trip of type T.
728 template<typename T, int B>
729 BlockSp::BlockSp<T, B> BlockSp_from_Eigen_Trip(const Eigen_Trip<T>& trip)
730 {
731 int m = trip.M; int n = trip.N; //the size of Eigen SparseMatrix
732 //Note the choice of block size B for the BlockSp matrix must divide
733 //the number of rows M and number of columns N
734 assert((m % B) == 0 && (n % B) == 0);
735 int lm = m / B; //number of rows in BlockSp matrix
736 int ln = n / B; //number of cols in BlockSp matrix
737 BlockSp::BlockSp<T, B> A(lm, ln);
738
739 //Fill BlockSp
740 for (const auto& tr : trip.m_trip)
741 BlockSp::util::setFromGlobal(A, tr.row(), tr.col(), tr.value());
742 return A;
743 }

766 ///Read in a matrix-market matrix file (.mtx) and convert it to an Eigen_Trip.
767 ///If symmetric, only half is in file, must make entire matrix.
768 ///If base_zero, the matrix indexing in the .mtx file starts from zero
769 template<typename T>
770 Eigen_Trip<T> mtxMatrix_to_Eigen_Trip(const std::string& mat_loc,
771 unsigned int start_line, bool symmetric, bool base_zero, bool printMatSize)
772 {
773 std::ifstream f(mat_loc, std::ios::in);
774
775 //Matrix market files are typically formatted in following manner:
776 //1) Name of matrix / matrix details.
777 //2) Number of rows, number of columns, number of entries/nonzeros.
778 //3) Values of matrix in matrix market format.
779 //Matrix market format: row index, col index, value.
780 //User must ensure start_line is the matrix size line (2).
781 BlockSp::tests::GotoLine(f, start_line);
782 std::string size; std::getline(f, size);
783 std::stringstream ss(size);
784 int M, N, nnz; ss >> M; ss >> N; ss >> nnz;
785 if (printMatSize)
786 {
787 std::cout << M << ' ' << N << ' ' << nnz;
788 if (symmetric) std::cout << " Symmetric";
789 std::cout << std::endl;
790 }
791
792 Eigen_Trip<T> trip; trip.M = M; trip.N = N;
793 for (int innz = 0; innz < nnz; ++innz)
794 {
795 std::string entry; std::getline(f, entry);
796 int i, j; T val;
797 std::stringstream ss2(entry);
798 ss2 >> i; ss2 >> j; //write matrix index and values.
799 if (!base_zero)
800 {
801 i -= 1; j -= 1; //change to base 0 indexing.
802 }
803 if constexpr (BlockSp::is_complex_v<T>) //also read imaginary value.
804 {
805 double valReal, valImg; ss2 >> valReal; ss2 >> valImg;
806 val.real(valReal); val.imag(valImg);
807 trip.push_back(Eigen::Triplet<T>(i, j, val));
808 if (symmetric && i != j) //if symmetric, fill (j,i) with complex conjugate.
809 trip.push_back(Eigen::Triplet<T>(j, i, std::conj(val)));
810 }
811 else
812 {
813 ss2 >> val;
814 trip.push_back(Eigen::Triplet<T>(i, j, val));
815 if (symmetric && i != j) //if symmetric, fill (j,i)
816 trip.push_back(Eigen::Triplet<T>(j, i, val));
817 }
818 }
819 return trip;
820 }
